export class HexIdGenerator {
    private app: App;

    constructor(app: App) {
        this.app = app;
    }

    // –ù–∞–π—Ç–∏ –ø–µ—Ä–≤—ã–π —Å–≤–æ–±–æ–¥–Ω—ã–π –º–∞—Å—Ç–µ—Ä ID (—É—Ä–æ–≤–µ–Ω—å 1)
    async findFreeMasterId(): Promise<string> {
        const files = this.app.vault.getMarkdownFiles();
        const masterFiles = files.filter(file => 
            /^[0-9A-F]{2} - /.test(file.basename)
        );

        const ids = masterFiles.map(file => {
            const match = file.basename.match(/^([0-9A-F]{2})/);
            return match ? parseInt(match[1], 16) : 0;
        }).sort((a, b) => a - b);

        let expected = 0;
        for (const id of ids) {
            if (id > expected) break;
            expected = id + 1;
        }

        return expected.toString(16).toUpperCase().padStart(2, '0');
    }

    // –ù–∞–π—Ç–∏ —Å–≤–æ–±–æ–¥–Ω—ã–π –¥–æ—á–µ—Ä–Ω–∏–π ID
    async findFreeChildId(parentId: string): Promise<string> {
        const pattern = new RegExp(\`^${parentId}-[0-9A-F]{2}\`);
        const files = this.app.vault.getMarkdownFiles();
        
        const childFiles = files.filter(file => 
            pattern.test(file.basename)
        );

        const ids = childFiles.map(file => {
            const match = file.basename.match(new RegExp(\`${parentId}-([0-9A-F]{2})\`));
            return match ? parseInt(match[1], 16) : 0;
        }).sort((a, b) => a - b);

        let expected = 1;
        for (const id of ids) {
            if (id > expected) break;
            expected = id + 1;
        }

        return expected.toString(16).toUpperCase().padStart(2, '0');
    }
}
