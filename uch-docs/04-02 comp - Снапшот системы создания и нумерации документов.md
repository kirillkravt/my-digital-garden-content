### 17.1. –î–µ–π—Å—Ç–≤–∏–µ: –°–æ–∑–¥–∞–µ–º –Ω–æ–≤—ã–π —Å–∫—Ä–∏–ø—Ç convert-fix.sh

```
#!/bin/bash

echo "=== UCH CONVERT-FIX: –ò–°–ü–†–ê–í–õ–ï–ù–ò–ï –î–û–ö–£–ú–ï–ù–¢–û–í –ò –°–°–´–õ–û–ö ==="

# –§—É–Ω–∫—Ü–∏—è –¥–ª—è –∏–∑–≤–ª–µ—á–µ–Ω–∏—è ID –∏–∑ –∏–º–µ–Ω–∏ —Ñ–∞–π–ª–∞
extract_id_from_filename() {
    local filename="$1"
    
    if [[ "$filename" =~ ^([0-9A-Fa-f]{2}(-[0-9A-Fa-f]{2})*) ]]; then
        echo "${BASH_REMATCH[1]}"
    else
        echo ""
    fi
}

# –§—É–Ω–∫—Ü–∏—è –¥–ª—è –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è —É—Ä–æ–≤–Ω—è –ø–æ ID
get_level_from_id() {
    local id="$1"
    if [ -z "$id" ]; then
        echo "0"
        return
    fi
    level=$(echo "$id" | tr -cd '-' | wc -c)
    echo $((level + 1))
}

# –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø–æ–∏—Å–∫–∞ –¥–æ—á–µ—Ä–Ω–∏—Ö –¥–æ–∫—É–º–µ–Ω—Ç–æ–≤
find_child_documents() {
    local parent_id="$1"
    
    if [ -z "$parent_id" ]; then
        echo ""
        return
    fi
    
    # –ò—â–µ–º –≤—Å–µ —Ñ–∞–π–ª—ã, –∫–æ—Ç–æ—Ä—ã–µ –Ω–∞—á–∏–Ω–∞—é—Ç—Å—è —Å parent_id-
    child_files=$(find . -maxdepth 1 -name "${parent_id}-*.md" -type f 2>/dev/null | sort)
    echo "$child_files"
}

# –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø–æ–∏—Å–∫–∞ —Ä–æ–¥–∏—Ç–µ–ª—å—Å–∫–æ–≥–æ –¥–æ–∫—É–º–µ–Ω—Ç–∞
find_parent_document() {
    local child_id="$1"
    
    if [ -z "$child_id" ]; then
        echo ""
        return
    fi
    
    # –û–ø—Ä–µ–¥–µ–ª—è–µ–º parent_id (—É–±–∏—Ä–∞–µ–º –ø–æ—Å–ª–µ–¥–Ω–∏–π —Å–µ–≥–º–µ–Ω—Ç)
    parent_id=$(echo "$child_id" | sed 's/-[^-]*$//')
    
    # –ò—â–µ–º —Ä–æ–¥–∏—Ç–µ–ª—å—Å–∫–∏–π —Ñ–∞–π–ª
    parent_file=$(find . -maxdepth 1 -name "${parent_id} - *.md" -type f | head -1)
    echo "$parent_file"
}

# –§—É–Ω–∫—Ü–∏—è –¥–ª—è –¥–æ–±–∞–≤–ª–µ–Ω–∏—è/–æ–±–Ω–æ–≤–ª–µ–Ω–∏—è —Ä–∞–∑–¥–µ–ª–∞ –î–û–ß–ï–†–ù–ò–ï –î–û–ö–£–ú–ï–ù–¢–´
update_children_section() {
    local file="$1"
    local fix_mode="$2"
    
    echo "  –ü—Ä–æ–≤–µ—Ä–∫–∞ —Ä–∞–∑–¥–µ–ª–∞ –î–û–ß–ï–†–ù–ò–ï –î–û–ö–£–ú–ï–ù–¢–´ –≤: $(basename "$file")"
    
    # –ò–∑–≤–ª–µ–∫–∞–µ–º ID —Ñ–∞–π–ª–∞
    filename=$(basename "$file" .md)
    file_id=$(extract_id_from_filename "$filename")
    
    if [ -z "$file_id" ]; then
        echo "    ‚ö†Ô∏è  –ù–µ —É–¥–∞–ª–æ—Å—å –∏–∑–≤–ª–µ—á—å ID, –ø—Ä–æ–ø—É—Å–∫–∞—é"
        return 1
    fi
    
    # –ò—â–µ–º –¥–æ—á–µ—Ä–Ω–∏–µ –¥–æ–∫—É–º–µ–Ω—Ç—ã
    child_files=$(find_child_documents "$file_id")
    
    if [ -n "$child_files" ]; then
        child_count=$(echo "$child_files" | wc -l | tr -d ' ')
        echo "    –ù–∞–π–¥–µ–Ω–æ –¥–æ—á–µ—Ä–Ω–∏—Ö –¥–æ–∫—É–º–µ–Ω—Ç–æ–≤: $child_count"
        
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º –Ω–∞–ª–∏—á–∏–µ —Ä–∞–∑–¥–µ–ª–∞ –î–û–ß–ï–†–ù–ò–ï –î–û–ö–£–ú–ï–ù–¢–´
        if grep -q "#### –î–û–ß–ï–†–ù–ò–ï –î–û–ö–£–ú–ï–ù–¢–´" "$file"; then
- [[00-04-02-01 - –ó–∞–¥–∞—á–∞. –°–æ–∑–¥–∞–Ω–∏–µ –¥–æ–∫—É–º–µ–Ω—Ç–∞ —Å —É–∫–∞–∑–∞–Ω–∏–µ–º –ø–æ–ª–Ω–æ–≥–æ –∏–º–µ–Ω–∏]]
- [[00-04-02-02 - –ó–∞–¥–∞—á–∞. –°–æ–∑–¥–∞—Ç—å —à–∞–±–ª–æ–Ω –¥–ª—è –∑–∞–¥–∞—á task]]
- [[00-04-02-03 - –ó–∞–¥–∞—á–∞. –î–æ–±–∞–≤–∏—Ç—å –≤–æ–∑–º–æ–∂–Ω–æ—Å—Ç—å —Å–æ–∑–¥–∞–≤–∞—Ç—å –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ –Ω–µ—Å–∫–æ–ª—å–∫–æ –¥–æ–∫—É–º–µ–Ω—Ç–æ–≤]]
- [[00-04-02-04 - –ó–∞–¥–∞—á–∞. –ó–∞–ø—É—Å–∫ —Å–∫—Ä–∏–ø—Ç–∞ –∏–∑ –ø–∞–ª–µ—Ç–∫–∏ –∫–æ–º–∞–Ω–¥ Obsidian]]
- [[00-04-02-05 - –ó–∞–¥–∞—á–∞. –ò—Å–ø—Ä–∞–≤–∏—Ç—å –±–∞–≥ —É–∫–∞–∑–∞–Ω–∏—è —Ç–µ–≥–æ–≤ –ø—Ä–∏ —Å–æ–∑–¥–∞–Ω–∏–∏ –¥–æ–∫—É–º–µ–Ω—Ç–æ–≤]]
- [[00-04-02-06 - –ó–∞–¥–∞—á–∞. –°–æ–∑–¥–∞–Ω–∏–µ –¥–æ–∫—É–º–µ–Ω—Ç–∞ —Å –Ω–∞–∏–≤—ã—Å—à—ã–º id —É—Ä–æ–≤–Ω—è]]
- [[00-04-02-07 - –ó–∞–¥–∞—á–∞. –§—É–Ω–∫—Ü–∏–æ–Ω–∞–ª—å–Ω–æ—Å—Ç—å –∑–∞–º–µ–Ω—ã –¥–æ–∫—É–º–µ–Ω—Ç–∞]]
- [[04-02-0A - –ê–Ω–∞–ª–∏–∑ —Å–æ—Å—Ç–æ—è–Ω–∏—è –ø—Ä–æ–µ–∫—Ç–∞]]
- [[04-02-01 - –ü–µ—Ä–µ–∏–º–µ–Ω–æ–≤–∞–Ω–∏–µ —Å —É–∫–∞–∑–∞–Ω–∏–µ–º id]]
- [[04-02-02 - –ü–µ—Ä–µ–∏–º–µ–Ω–æ–≤–∞–Ω–∏–µ —Å –¥–æ–±–∞–≤–ª–µ–Ω–∏–µ–º id]]
            echo "    ‚úÖ –†–∞–∑–¥–µ–ª –î–û–ß–ï–†–ù–ò–ï –î–û–ö–£–ú–ï–ù–¢–´ —Å—É—â–µ—Å—Ç–≤—É–µ—Ç"
            
            # –°—á–∏—Ç–∞–µ–º —Å—É—â–µ—Å—Ç–≤—É—é—â–∏–µ —Å—Å—ã–ª–∫–∏
            existing_links=$(grep -c "\[\[$file_id-" "$file" 2>/dev/null || echo "0")
            
            if [ "$existing_links" -ne "$child_count" ]; then
                echo "    ‚ö†Ô∏è  –†–∞—Å—Ö–æ–∂–¥–µ–Ω–∏–µ: —Å—Å—ã–ª–æ–∫ $existing_links, –¥–æ—á–µ—Ä–Ω–∏—Ö $child_count"
                
                if [ "$fix_mode" = "true" ]; then
                    echo "    –ò—Å–ø—Ä–∞–≤–ª—è—é..."
                    
                    # –°–æ–∑–¥–∞–µ–º –≤—Ä–µ–º–µ–Ω–Ω—ã–π —Ñ–∞–π–ª
                    temp_file="${file}.tmp"
                    
                    # –ü–µ—Ä–µ—Å–æ–∑–¥–∞–µ–º —Ä–∞–∑–¥–µ–ª —Å –ø—Ä–∞–≤–∏–ª—å–Ω—ã–º–∏ —Å—Å—ã–ª–∫–∞–º–∏
                    awk -v child_files="$child_files" '
                    BEGIN {
                        split(child_files, files, "\n")
                        in_children_section = 0
                        section_rebuilt = 0
                    }
                    
                    /#### –î–û–ß–ï–†–ù–ò–ï –î–û–ö–£–ú–ï–ù–¢–´/ {
                        in_children_section = 1
                        section_rebuilt = 1
                        print $0
                        # –î–æ–±–∞–≤–ª—è–µ–º –≤—Å–µ –∞–∫—Ç—É–∞–ª—å–Ω—ã–µ —Å—Å—ã–ª–∫–∏
                        for (i in files) {
                            if (files[i] != "") {
                                split(files[i], parts, "/")
                                child_filename = parts[length(parts)]
                                gsub(/\.md$/, "", child_filename)
                                # –ü—Ä–æ–≤–µ—Ä—è–µ–º, —á—Ç–æ —ç—Ç–æ –¥–µ–π—Å—Ç–≤–∏—Ç–µ–ª—å–Ω–æ –¥–æ—á–µ—Ä–Ω–∏–π –¥–æ–∫—É–º–µ–Ω—Ç
                                if (child_filename ~ "^" file_id "-") {
                                    print "- [[" child_filename "]]"
                                }
                            }
                        }
                        next
                    }
                    
                    in_children_section && /^[[:space:]]*–ü–æ–∫–∞ –Ω–µ—Ç –¥–æ—á–µ—Ä–Ω–∏—Ö –¥–æ–∫—É–º–µ–Ω—Ç–æ–≤/ {
                        # –ü—Ä–æ–ø—É—Å–∫–∞–µ–º —ç—Ç—É —Å—Ç—Ä–æ–∫—É
                        next
                    }
                    
                    in_children_section && (/^####/ || /^###/ || /^##/ || /^#/ || /^---/) {
                        in_children_section = 0
                    }
                    
                    {
                        if (!in_children_section) {
                            print $0
                        }
                    }
                    ' file_id="$file_id" "$file" > "$temp_file"
                    
                    mv "$temp_file" "$file"
                    echo "    ‚úÖ –†–∞–∑–¥–µ–ª –∏—Å–ø—Ä–∞–≤–ª–µ–Ω"
                fi
            else
                echo "    ‚úÖ –í—Å–µ —Å—Å—ã–ª–∫–∏ –∫–æ—Ä—Ä–µ–∫—Ç–Ω—ã"
            fi
        else
            echo "    ‚ö†Ô∏è  –†–∞–∑–¥–µ–ª –î–û–ß–ï–†–ù–ò–ï –î–û–ö–£–ú–ï–ù–¢–´ –æ—Ç—Å—É—Ç—Å—Ç–≤—É–µ—Ç"
            
            if [ "$fix_mode" = "true" ]; then
                echo "    –î–æ–±–∞–≤–ª—è—é —Ä–∞–∑–¥–µ–ª..."
                
                # –î–æ–±–∞–≤–ª—è–µ–º —Ä–∞–∑–¥–µ–ª –ø–µ—Ä–µ–¥ —Ä–∞–∑–¥–µ–ª–∏—Ç–µ–ª–µ–º --- –∏–ª–∏ –≤ –∫–æ–Ω–µ—Ü —Ñ–∞–π–ª–∞

                        split(child_files, files, "\n")
                        children_added = 0

                                    }
                                }
                            }

                            next
                        }
                    }

                    ' "$file" > "${file}.tmp"
                else
                    # –î–æ–±–∞–≤–ª—è–µ–º –≤ –∫–æ–Ω–µ—Ü —Ñ–∞–π–ª–∞
                    echo "" >> "$file"

                        child_name=$(basename "$child" .md)
                        echo "- [[$child_name]]" >> "$file"
                    done
                fi
                
                echo "    ‚úÖ –†–∞–∑–¥–µ–ª –¥–æ–±–∞–≤–ª–µ–Ω"
            fi
        fi

        # –ü—Ä–æ–≤–µ—Ä—è–µ–º, –µ—Å—Ç—å –ª–∏ —Ä–∞–∑–¥–µ–ª —Å "–ü–æ–∫–∞ –Ω–µ—Ç –¥–æ—á–µ—Ä–Ω–∏—Ö –¥–æ–∫—É–º–µ–Ω—Ç–æ–≤"

            if [ "$fix_mode" = "true" ]; then
                echo "    –î–æ–±–∞–≤–ª—è—é —Ä–∞–∑–¥–µ–ª..."
                
                if grep -q "^---" "$file"; then
                    awk '
                    BEGIN {
                        children_added = 0
                        in_frontmatter = 0
                        frontmatter_end = 0
                    }
                    
                    /^---/ {
                        frontmatter_end++
                        if (frontmatter_end == 2 && !children_added) {
                            print $0
                            print ""
                            print "#### –î–û–ß–ï–†–ù–ò–ï –î–û–ö–£–ú–ï–ù–¢–´"
                            print "–ü–æ–∫–∞ –Ω–µ—Ç –¥–æ—á–µ—Ä–Ω–∏—Ö –¥–æ–∫—É–º–µ–Ω—Ç–æ–≤."
                            children_added = 1
                            next
                        }
                    }
                    { print $0 }

                else
```
